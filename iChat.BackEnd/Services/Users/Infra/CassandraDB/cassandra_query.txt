CREATE TABLE IF NOT EXISTS db_user_message.messages (
    channel_id BIGINT,
    message_id BIGINT,
    sender_id BIGINT,
    message_type SMALLINT,
    text_content TEXT,
    media_content TEXT,
    timestamp TIMESTAMP,
    PRIMARY KEY ((channel_id), message_id, sender_id)
) WITH CLUSTERING ORDER BY (message_id DESC,sender_id ASEC);
